API = "https://en.wikipedia.org/w/api.php"


def fetch_full_wikitext(title: str) -> str:
    r = httpx.get(API, params={
        "action": "parse", "format": "json", "page": title,
        "prop": "wikitext", "redirects": "1",
    }, headers={"User-Agent": UA}, timeout=20)
    r.raise_for_status()
    data = r.json()
    return "" if "error" in data else data.get("parse", {}).get("wikitext", {}).get("*", "")


def fetch_lead(title: str, exchars: int = 2000) -> str:
    r = httpx.get(API, params={
        "action": "query", "format": "json", "prop": "extracts",
        "exintro": "1", "explaintext": "1", "exchars": str(exchars),
        "titles": title, "redirects": "1",
    }, headers={"User-Agent": UA}, timeout=15)
    r.raise_for_status()
    pages = r.json().get("query", {}).get("pages", {})
    page = next(iter(pages.values()))
    return page.get("extract", "")


def context(text: str, needle: str, window: int = 200) -> str:
    idx = text.lower().find(needle.lower())
    if idx < 0:
        return f"<no match for {needle!r}>"
    return text[max(0, idx-window):idx+len(needle)+window]


def section(label: str) -> None:
    print(f"\n{'=' * 70}\n{label}\n{'=' * 70}")
